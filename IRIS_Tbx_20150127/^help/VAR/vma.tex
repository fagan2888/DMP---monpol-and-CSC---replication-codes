

    \filetitle{vma}{Matrices describing the VMA representation of a VAR process}{VAR/vma}

	\paragraph{Syntax}

\begin{verbatim}
Phi = vma(V,N)
\end{verbatim}

\paragraph{Input arguments}

\begin{itemize}
\item
  \texttt{V} {[} VAR {]} - VAR object for which the VMA matrices will be
  computed.
\item
  \texttt{N} {[} numeric {]} - Order up to which the VMA matrices will
  be computed.
\end{itemize}

\paragraph{Output arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{Phi} {[} numeric {]} - VMA matrices.
\end{itemize}
