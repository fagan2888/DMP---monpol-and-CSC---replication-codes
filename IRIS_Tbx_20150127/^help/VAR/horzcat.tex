

    \filetitle{horzcat}{Combine two compatible VAR objects in one object with multiple parameterisations}{VAR/horzcat}

	\paragraph{Syntax}

\begin{verbatim}
V = [V1,V2,...]
\end{verbatim}

\paragraph{Input arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{V1}, \texttt{V2} {[} VAR {]} - Compatible VAR objects that
  will be combined.
\end{itemize}

\paragraph{Output arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{V} {[} VAR {]} - Output VAR object that combines the input VAR
  objects as multiple parameterisations.
\end{itemize}
