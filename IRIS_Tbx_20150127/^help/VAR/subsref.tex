

    \filetitle{subsref}{Subscripted reference for VAR objects}{VAR/subsref}

	\paragraph{Syntax to retrieve VAR object with subset of
parameterisations}

\begin{verbatim}
V(Inx)
\end{verbatim}

\paragraph{Input arguments}

\begin{itemize}
\item
  \texttt{V} {[} VAR {]} - VAR object.
\item
  \texttt{Inx} {[} numeric \textbar{} logical {]} - Index of requested
  parameterisations.
\end{itemize}
