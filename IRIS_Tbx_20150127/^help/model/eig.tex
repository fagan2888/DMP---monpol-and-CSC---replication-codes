

    \filetitle{eig}{Eigenvalues of the transition matrix}{model/eig}

	\paragraph{Syntax}

\begin{verbatim}
e = eig(m)
\end{verbatim}

\paragraph{Input arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{m} {[} model {]} - Model object whose eigenvalues will be
  returned.
\end{itemize}

\paragraph{Output arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{e} {[} numeric {]} - Array of all eigenvalues associated with
  the model, i.e.~all stable, unit, and unstable roots are included.
\end{itemize}
