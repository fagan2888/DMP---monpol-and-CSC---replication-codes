

    \filetitle{transpose}{Transpose each page of matrix with names rows and columns}{namedmat/transpose}

	\paragraph{Syntax}

\begin{verbatim}
X = transpose(X)
X = X.'
\end{verbatim}

\paragraph{Input arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{X} {[} namedmat {]} - Input matrix or array with named rows
  and columns.
\end{itemize}

\paragraph{Output arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{X} {[} namedmat {]} - Transpose of the input matrix; if it is
  more than 2-dimensional, each page of the matrix is transposed.
\end{itemize}
