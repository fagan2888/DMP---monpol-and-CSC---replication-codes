

    \filetitle{dat2charlist}{Convert dates to a comma-separated list}{dates/dat2charlist}

	\paragraph{Syntax}

\begin{verbatim}
C = dat2charlist(D,...)
\end{verbatim}

\paragraph{Input arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{D} {[} numeric {]} - IRIS serial date numbers that will be
  converted to a comma-separated list.
\end{itemize}

\paragraph{Output arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{C} {[} char {]} - Text string with a comma-separated list of
  dates.
\end{itemize}

\paragraph{Options}

See help on \href{dates/dat2str}{\texttt{dat2str}} for options
available.
