

    \filetitle{mtimes}{Keep only the database entries that are on the list}{dbase/dbmtimes}

	\paragraph{Syntax}

\begin{verbatim}
D = D * List
\end{verbatim}

\paragraph{Input arguments}

\begin{itemize}
\item
  \texttt{D} {[} struct {]} - Input database.
\item
  \texttt{List} {[} cellstr {]} - List of entries that will be kept in
  the output database.
\end{itemize}

\paragraph{Output arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{D} {[} struct {]} - Output database where only the input
  entries that are in the \texttt{List} are included.
\end{itemize}
