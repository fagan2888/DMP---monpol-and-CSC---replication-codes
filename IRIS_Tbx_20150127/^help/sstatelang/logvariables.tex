

    \filetitle{!log\_variables}{Restrict the steady state of some of the variables to be positive}{sstatelang/logvariables}

	\paragraph{Syntax}
 
 \begin{verbatim}
 !log_variables
     Variable_Name, Variable_Name
     Variable_Name
 \end{verbatim}
 
 \paragraph{Syntax with inverted list}
 
 \begin{verbatim}
 !log_variables
     !all_but
     Variable_Name, Variable_Name
     Variable_Name
 \end{verbatim}
