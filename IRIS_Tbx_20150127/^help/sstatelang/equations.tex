

    \filetitle{!equations}{Block of equations or assignments}{sstatelang/equations}

	\paragraph{Syntax for a system of equations to be solved numerically}
 
 \begin{verbatim}
 !equations
     EQUATION;
     EQUATION;
     EQUATION;
     ...
 
     !solvefor
         LIST_OF_VARIABLES
 \end{verbatim}
 
 \paragraph{Syntax for a system of equations to be solved symbolically}
 
 \begin{verbatim}
 !equations
     EQUATION;
     EQUATION;
     EQUATION;
     ...
 
     !symbolic
     !solvefor
         LIST_OF_VARIABLES
 \end{verbatim}
 
 \paragraph{Syntax for block of assignments}
 
 \begin{verbatim}
 !equations
     VARIABLE_NAME = EXPRESSION;
     VARIABLE_NAME = EXPRESSION;
     VARIABLE_NAME = EXPRESSION;
     ...
 \end{verbatim}
