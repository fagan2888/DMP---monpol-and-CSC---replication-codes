

    \filetitle{empty}{Empty time series preserving the size in 2nd and higher dimensions}{tseries/empty}

	\paragraph{Syntax}

\begin{verbatim}
x = empty(x)
\end{verbatim}

\paragraph{Input arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{This} {[} tseries {]} - Input time series that will be
  emptied.
\end{itemize}

\paragraph{Output arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{This} {[} tseries {]} - Empty time series with the 2nd and
  higher dimensions the same size as the input tseries object, and
  comments preserved.
\end{itemize}
