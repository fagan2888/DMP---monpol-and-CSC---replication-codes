

    \filetitle{subsref}{Subscripted reference function for tseries objects}{tseries/subsref}

	\paragraph{Syntax returning numeric
array}

\begin{verbatim}
... = X(Dates)
... = X(Dates,...)
\end{verbatim}

\paragraph{Syntax returning tseries
object}

\begin{verbatim}
... = X{Dates}
... = X{Dates,...}
\end{verbatim}

\paragraph{Input arguments}

\begin{itemize}
\item
  \texttt{X} {[} tseries {]} - Tseries object.
\item
  \texttt{Dates} {[} numeric {]} - Dates for which the time series
  observations will be returned, either as a numeric array or as another
  tseries object.
\end{itemize}
