

    \filetitle{double}{Return tseries observations as double-precision numeric array}{tseries/double}

	\paragraph{Syntax}

\begin{verbatim}
y = double(x)
\end{verbatim}

\paragraph{Input arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{x} {[} tseries {]} - Tseries object whose observations will be
  returned as double-precision numeric array.
\end{itemize}

\paragraph{Output arguments}

\begin{itemize}
\itemsep1pt\parskip0pt\parsep0pt
\item
  \texttt{y} {[} numeric {]} - Double-precision numeric array with the
  input tseries observations in columns.
\end{itemize}
